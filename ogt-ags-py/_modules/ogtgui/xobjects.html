[docs]    def set_header(self, col, text, align=None, tooltip=None, sort=None):
        """Sets a horizontal header

        :param col: The column index
        :type col: int
        :param label: The text label
        :type label: str`
        :param align: Text alignment
        :type align: Qt.Align
        :param tooltip: The tooltip to show on header hover
        :type tooltip: str
        :param sort: # TBA
        :type sort: int

        """
        item = xwidgets.StandardItem()
        item.setText(text)
        if align:
            item.setTextAlignment(align)
        if tooltip:
            item.setToolTip(tooltip)
        if sort:
            item.setData(sort, DATA_ROLE)
        self.setHorizontalHeaderItem(col, item)

[docs]    def make_row(self, xid, col):
        arr = []
        for c in range(0, self.columnCount()):
            item = xwidgets.StandardItem()
            if c == col:
                item.setText(str(xid))
            arr.append(item)
        self.appendRow(arr)
        return arr[0].row()
